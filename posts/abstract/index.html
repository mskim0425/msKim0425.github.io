abstract class User {
    public abstract String name();
    public abstract String email();
}

class kakaoUser extends User{
    private String name;
    private String email;

    public kakaoUser(String name, String email) {
        this.name = name;
        this.email = email;
    } 
    
    @Override
    public String name() {
        return this.name;
    }

    @Override
    public String email() {
        return this.email;
    }
}


class googleUser extends User{
    private String name;
    private String email;

    public googleUser(String name, String email) {
        this.name = name;
        this.email = email;
    }

    @Override
    public String name() {
        return this.name;
    }

    @Override
    public String email() {
        return this.email;
    }
}
