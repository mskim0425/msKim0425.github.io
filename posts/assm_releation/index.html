class Order
  include AASM

  aasm do
    state :draft, initial: true
    state :confirmed
    state :shipped
    state :delivered
    
    event :confirm do
      transitions from: :draft, to: :confirmed
    end
    
    event :ship do
      transitions from: :confirmed, to: :shipped
    end
    
    event :deliver do
      transitions from: :shipped, to: :delivered
    end
  end
end
